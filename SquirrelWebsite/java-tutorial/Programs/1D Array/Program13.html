13. Program to right rotate the elements of an array

Code


                public class RightRotateArrayExample {
                    public static void main(String[] args) {
                        int[] array = {1, 2, 3, 4, 5};
                        int rotationCount = 2;
                                                        
                        System.out.println("Original Array:");
                        printArray(array);
                                                        
                        // Right rotate the array
                        rightRotateArray(array, rotationCount);
                                                        
                        System.out.println("\nArray after Right Rotation:");
                        printArray(array);
                    }
                                                        
                    private static void rightRotateArray(int[] array, int rotationCount) {
                        int length = array.length;
                                                        
                        // Perform right rotation
                        for (int i = 0; i < rotationCount; i++) {
                            int temp = array[length - 1];
                            for (int j = length - 1; j > 0; j--) {
                                array[j] = array[j - 1];
                            }
                            array[0] = temp;
                        }
                    }
                                                        
                    private static void printArray(int[] array) {
                        for (int value : array) {
                            System.out.print(value + " ");
                        }
                    }
                }

Output

Original Array: 

1 2 3 4 5  

 

Array after Right Rotation: 

4 5 1 2 3
