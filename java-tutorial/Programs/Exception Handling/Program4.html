Exception Handling

4. Write a Java program that reads a list of numbers from a file and throws an exception if any of the numbers are positive

Code


                
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.FileNotFoundException;
import java.io.IOException;
                                        
public class NegativeNumberCheck {
    public static void main(String[] args) {
        try {
            String fileName = "numbers.txt"; // Replace with the actual file name
            checkNumbers(fileName);
            System.out.println("All numbers are non-positive.");
        } catch (FileNotFoundException e) {
            System.out.println("Error: File not found - " + e.getMessage());
        } catch (IOException e) {
            System.out.println("Error reading the file - " + e.getMessage());
        } catch (PositiveNumberException e) {
            System.out.println("Error: Positive number detected - " + e.getMessage());
        }
    }
                                        
    private static void checkNumbers(String fileName) throws FileNotFoundException, IOException, PositiveNumberException {
        try (BufferedReader reader = new BufferedReader(new FileReader(fileName))) {
            String line;
                                        
            while ((line = reader.readLine()) != null) {
                int number = Integer.parseInt(line);
                                        
                if (number > 0) {
                    throw new PositiveNumberException("Positive number found: " + number);
                }
            }
        }
    }
}
                                        
class PositiveNumberException extends Exception {
    public PositiveNumberException(String message) {
        super(message);
}

Output

    All numbers are non-positive.
