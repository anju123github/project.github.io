AWT

1. Create a simple AWT application that displays a button with FlowLayout manager.

Code


                
import java.awt.*;
import java.awt.event.*;

public class SimpleAWTApplication {
    public static void main(String[] args) {
        // Create the frame
        Frame frame = new Frame("Simple AWT Application");

        // Create a button
        Button button = new Button("Click Me!");

        // Set the layout manager to FlowLayout
        frame.setLayout(new FlowLayout());

        // Add the button to the frame
        frame.add(button);

        // Add an action listener to the button
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                // Print a message to the console when the button is clicked
                System.out.println("Button Clicked!");
            }
        });

        // Set the frame size
        frame.setSize(300, 200);

        // Make the frame visible
        frame.setVisible(true);

        // Add a window listener to handle closing the frame
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }
}
