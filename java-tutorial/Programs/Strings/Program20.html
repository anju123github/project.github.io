20. Program to find all subsets of a string

Code


                
import java.util.ArrayList;
import java.util.Scanner;

public class SubsetsOfString {

    // Function to find all subsets of a string
    static ArrayList getAllSubsets(String str) {
        int n = str.length();
        ArrayList subsets = new ArrayList<>();

        // Total number of subsets is 2^n
        int totalSubsets = (int) Math.pow(2, n);

        // Generate all possible binary representations of subsets
        for (int i = 0; i < totalSubsets; i++) {
            StringBuilder subset = new StringBuilder();

            // Map each bit of binary representation to corresponding character in the string
            for (int j = 0; j < n; j++) {
                if ((i & (1 << j)) != 0) {
                    subset.append(str.charAt(j));
                }
            }

            // Add the subset to the list
            subsets.add(subset.toString());
        }

        return subsets;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Input the string
        System.out.print("Enter the string: ");
        String inputString = scanner.nextLine();

        // Get all subsets of the string
        ArrayList subsets = getAllSubsets(inputString);

        // Display the subsets
        System.out.println("All subsets of the string are:");
        for (String subset : subsets) {
            System.out.println(subset);
        }

        scanner.close();
    }
}
