11. Program to determine whether a given matrix is an identity matrix

Code


             
import java.util.Scanner;

public class IdentityMatrix {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
                                
        // Get the order of the square matrix
        System.out.print("Enter the order of the square matrix: ");
        int order = scanner.nextInt();
                                
        // Create a square matrix
        int[][] matrix = new int[order][order];
                                
        // Get elements for the matrix
        enterMatrixElements(scanner, matrix);
                                
        // Check if the matrix is an identity matrix
        boolean isIdentityMatrix = checkIdentityMatrix(matrix);
                                
        // Print the original matrix and the result
        System.out.println("Original Matrix:");
        printMatrix(matrix);
                                
        if (isIdentityMatrix) {
            System.out.println("The matrix is an Identity Matrix.");
        } else {
            System.out.println("The matrix is not an Identity Matrix.");
        }
                                
        scanner.close();
    }
                                
    // Function to enter matrix elements
    private static void enterMatrixElements(Scanner scanner, int[][] matrix) {
        System.out.println("Enter elements for the matrix:");
                                
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print("Enter element at position [" + i + "][" + j + "]: ");
                matrix[i][j] = scanner.nextInt();
            }
        }
    }
                                
    // Function to check if a matrix is an identity matrix
    private static boolean checkIdentityMatrix(int[][] matrix) {
        int order = matrix.length;
                                
        // Check if the matrix is square
        if (matrix[0].length != order) {
            return false;
        }
                                
        // Check the diagonal elements
        for (int i = 0; i < order; i++) {
            for (int j = 0; j < order; j++) {
                if (i == j && matrix[i][j] != 1) {
                    return false;
                } else if (i != j && matrix[i][j] != 0) {
                    return false;
                }
            }
        }
                                
        return true;
    }
                                
    // Function to print a matrix
    private static void printMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println(); // Move to the next row
        }
        System.out.println(); // Separate matrices by a blank line
    }
}
