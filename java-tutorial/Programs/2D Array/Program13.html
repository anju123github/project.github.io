13. Program to determine whether a given matrix is a sparse matrix

Code


                
import java.util.Scanner;

public class SparseMatrix {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
                                
        // Get the dimensions of the matrix
        System.out.print("Enter the number of rows: ");
        int rows = scanner.nextInt();
                                
        System.out.print("Enter the number of columns: ");
        int columns = scanner.nextInt();
                                
        // Create a matrix
        int[][] matrix = new int[rows][columns];
                                
        // Get elements for the matrix
        enterMatrixElements(scanner, matrix);
                                
        // Check if the matrix is a sparse matrix
        boolean isSparseMatrix = checkSparseMatrix(matrix);
                                
        // Print the original matrix and the result
        System.out.println("Original Matrix:");
        printMatrix(matrix);
                                
        if (isSparseMatrix) {
            System.out.println("The matrix is a Sparse Matrix.");
        } else {
            System.out.println("The matrix is not a Sparse Matrix.");
        }
                                
        scanner.close();
    }
                                
    // Function to enter matrix elements
    private static void enterMatrixElements(Scanner scanner, int[][] matrix) {
        System.out.println("Enter elements for the matrix:");
                                
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print("Enter element at position [" + i + "][" + j + "]: ");
                matrix[i][j] = scanner.nextInt();
            }
        }
    }
                                
    // Function to check if a matrix is a sparse matrix
    private static boolean checkSparseMatrix(int[][] matrix) {
        int zeroCount = 0;
        int nonZeroCount = 0;
                                
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                if (matrix[i][j] == 0) {
                    zeroCount++;
                } else {
                    nonZeroCount++;
                }
            }
        }
                                
        // If the number of zero elements is greater than or equal to non-zero elements,
        // then the matrix is considered sparse.
        return zeroCount >= nonZeroCount;
    }
                                
    // Function to print a matrix
    private static void printMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println(); // Move to the next row
        }
        System.out.println(); // Separate matrices by a blank line
    }
}
